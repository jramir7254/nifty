```sql
-- UUID generation (common in Neon/Postgres)
create extension if not exists pgcrypto;

create table users (
    id uuid primary key default gen_random_uuid(),
    email text unique not null,
    name text,
    created_at timestamptz not null default now()
);

create table assignments (
    id uuid primary key default gen_random_uuid(),
    created_by_user_id uuid not null references users(id),
    title text,
    created_at timestamptz not null default now(),
    current_version_id uuid
);

create table assignment_versions (
    id uuid primary key default gen_random_uuid(),
    assignment_id uuid not null references assignments(id) on delete cascade,
    parent_version_id uuid references assignment_versions(id),
    created_by_user_id uuid not null references users(id),
    source text not null default 'generated',
    created_at timestamptz not null default now(),

    params jsonb not null,
    content jsonb not null,

    -- optional: fast display / search
    content_text text,

    -- model/prompt/token usage/etc.
    gen_metadata jsonb not null default '{}'::jsonb
);

alter table assignments
add constraint fk_current_version
foreign key (current_version_id) references assignment_versions(id);

create table assignment_ratings (
    id uuid primary key default gen_random_uuid(),
    assignment_version_id uuid not null references assignment_versions(id) on delete cascade,
    user_id uuid not null references users(id),
    rating int not null check (rating between 1 and 5),
    created_at timestamptz not null default now(),
    unique (assignment_version_id, user_id)
);

create table assignment_comments (
    id uuid primary key default gen_random_uuid(),
    assignment_version_id uuid not null references assignment_versions(id) on delete cascade,
    user_id uuid not null references users(id),
    body text not null,
    created_at timestamptz not null default now()
);

create table assignment_refinements (
    id uuid primary key default gen_random_uuid(),
    assignment_version_id uuid not null references assignment_versions(id) on delete cascade,
    requested_by_user_id uuid not null references users(id),
    created_at timestamptz not null default now(),
    type text not null default 'refinement',
    payload jsonb not null default '{}'::jsonb,
    status text not null default 'open',
    result_version_id uuid references assignment_versions(id)
);

-- Indexes you’ll actually use
create index idx_versions_assignment_created
on assignment_versions (assignment_id, created_at desc);

create index idx_versions_created_by_created
on assignment_versions (created_by_user_id, created_at desc);

-- JSONB query acceleration (for filtering by params keys)
create index idx_versions_params_gin
on assignment_versions using gin (params);

create index idx_versions_metadata_gin
on assignment_versions using gin (gen_metadata);
```
